-- Удаляем старые функции DROP FUNCTION IF EXISTS register_user; DROP FUNCTION IF EXISTS login_user; DROP FUNCTION IF EXISTS create_app; DROP FUNCTION IF EXISTS increment_views; -- Простая функция регистрации CREATE OR REPLACE FUNCTION register_user(username TEXT, user_password TEXT) RETURNS JSON AS $$ DECLARE new_user_id INTEGER; BEGIN INSERT INTO crs_users (username, password) VALUES (username, user_password) RETURNING id INTO new_user_id; RETURN json_build_object('success', true, 'user_id', new_user_id); EXCEPTION WHEN unique_violation THEN RETURN json_build_object('success', false, 'error', 'Пользователь уже существует'); END; $$ LANGUAGE plpgsql; -- Простая функция логина CREATE OR REPLACE FUNCTION login_user(username TEXT, user_password TEXT) RETURNS JSON AS $$ DECLARE user_record crs_users%ROWTYPE; BEGIN SELECT * INTO user_record FROM crs_users WHERE crs_users.username = username AND crs_users.password = user_password; IF user_record.id IS NOT NULL THEN RETURN json_build_object('success', true, 'user_id', user_record.id, 'username', user_record.username); ELSE RETURN json_build_object('success', false, 'error', 'Неверные имя или пароль'); END IF; END; $$ LANGUAGE plpgsql; -- Функция создания приложения CREATE OR REPLACE FUNCTION create_app(app_title TEXT, app_code TEXT, author_id INTEGER, public BOOLEAN DEFAULT true) RETURNS JSON AS $$ DECLARE new_app_id INTEGER; BEGIN INSERT INTO crs_apps (title, code, author_id, is_public) VALUES (app_title, app_code, author_id, public) RETURNING id INTO new_app_id; RETURN json_build_object('success', true, 'app_id', new_app_id); END; $$ LANGUAGE plpgsql; -- Проверяем существующих пользователей SELECT id, username, password FROM crs_users;
